# Sri Murugan Store — Master Bug Fix Prompt

## ROLE
Fix specific bugs in the existing application only. No page rewrites. No architecture changes. Apply surgical fixes. Rebuild backend after every change: `cd backend && npm run build`

## ABSOLUTE RULES
- All inputs: inline styles only — no Tailwind classes on form/input elements
- Never define sub-components inside parent functions (destroys focus on every keystroke)
- Never auto-fill "Actual" fields from "Expected" values — they are intentionally separate
- Always compute financial totals server-side from DB — never trust client-sent numbers
- Use `ALTER TABLE … ADD COLUMN IF NOT EXISTS` — never DROP tables


### 4c — Backend: POST /cashouts uses upsert (allows re-edit same day)
```ts
export async function createOrUpdateCashout(req: Request, res: Response) {
  const { actual_cash, actual_gpay, actual_card, notes } = req.body;

  // Always fetch expected from DB — never trust client
  const s = await pool.query(`
    SELECT
      COALESCE(SUM(grand_total) FILTER (WHERE payment_mode='cash'   AND payment_status='paid'),0) AS expected_cash,
      COALESCE(SUM(grand_total) FILTER (WHERE payment_mode='upi'    AND payment_status='paid'),0) AS expected_upi,
      COALESCE(SUM(grand_total) FILTER (WHERE payment_mode='card'   AND payment_status='paid'),0) AS expected_card,
      COALESCE(SUM(grand_total) FILTER (WHERE payment_mode='credit' AND payment_status='paid'),0) AS expected_credit
    FROM bills WHERE DATE(created_at) = CURRENT_DATE
  `).then(r => r.rows[0]);

  const cashout = await pool.query(`
    INSERT INTO cashouts
      (cashier_id, cashout_date, expected_cash, expected_upi, expected_card, expected_credit,
       actual_cash, actual_gpay, actual_card, notes, status, closed_at)
    VALUES ($1, CURRENT_DATE, $2, $3, $4, $5, $6, $7, $8, $9, 'closed', NOW())
    ON CONFLICT (cashout_date) DO UPDATE SET
      actual_cash  = EXCLUDED.actual_cash,
      actual_gpay  = EXCLUDED.actual_gpay,
      actual_card  = EXCLUDED.actual_card,
      notes        = EXCLUDED.notes,
      closed_at    = NOW(),
      cashier_id   = EXCLUDED.cashier_id,
      -- Refresh expected values too (in case more bills came in since last save):
      expected_cash   = EXCLUDED.expected_cash,
      expected_upi    = EXCLUDED.expected_upi,
      expected_card   = EXCLUDED.expected_card,
      expected_credit = EXCLUDED.expected_credit
    RETURNING *
  `, [req.user.id, s.expected_cash, s.expected_upi, s.expected_card, s.expected_credit,
      actual_cash ?? null, actual_gpay ?? null, actual_card ?? null, notes ?? null]);

  await pool.query(
    `INSERT INTO audit_logs (action, performed_by, entity_type, entity_id, new_values)
     VALUES ('CASHOUT', $1, 'cashout', $2, $3)`,
    [req.user.id, cashout.rows[0].id, JSON.stringify(cashout.rows[0])]
  );

  res.json(cashout.rows[0]);
}
```

### 4d — Frontend Cashout.tsx — full rewrite of state and UI logic

**CRITICAL: Use flat individual useState. Never spread a form object. Never auto-fill actual from expected.**

```tsx
// State — flat, individual, never auto-populated from summary:
const [actualCash, setActualCash] = useState('');   // ← NEVER setActualCash(summary.cash_total)
const [actualGpay, setActualGpay] = useState('');   // ← NEVER setActualGpay(summary.gpay_total)
const [notes,      setNotes]      = useState('');
const [summary,    setSummary]    = useState<any>(null);
const [existing,   setExisting]   = useState<any>(null);
const [isEditing,  setIsEditing]  = useState(false);
const [loading,    setLoading]    = useState(false);
const [fetching,   setFetching]   = useState(false);

// Computed: final cashout = sum of actual cash + actual gpay
const finalCashout = (Number(actualCash) || 0) + (Number(actualGpay) || 0);

async function fetchSummary() {
  setFetching(true);
  try {
    const [sumRes, existRes] = await Promise.all([
      axios.get('/api/v1/cashouts/today-summary'),
      axios.get('/api/v1/cashouts?date=today'),
    ]);
    setSummary(sumRes.data);
    if (existRes.data.length > 0) {
      const c = existRes.data[0];
      setExisting(c);
      // Pre-fill actuals from existing cashout record (not from expected):
      setActualCash(c.actual_cash ?? '');
      setActualGpay(c.actual_gpay ?? '');
      setNotes(c.notes ?? '');
      setIsEditing(false);
    } else {
      setExisting(null);
      setIsEditing(true);  // new cashout — show form immediately
    }
  } finally { setFetching(false); }
}

useEffect(() => { fetchSummary(); }, []);

async function handleSubmit(e: React.FormEvent) {
  e.preventDefault();
  setLoading(true);
  try {
    await axios.post('/api/v1/cashouts', {
      actual_cash: actualCash !== '' ? Number(actualCash) : null,
      actual_gpay: actualGpay !== '' ? Number(actualGpay) : null,
      notes,
    });
    await fetchSummary();  // reload after submit
    setIsEditing(false);
    toast.success('Cashout saved');
  } catch (err: any) {
    toast.error(err?.response?.data?.message || 'Failed to save cashout');
  } finally { setLoading(false); }
}
```

```ts
export async function getTopProducts(req: Request, res: Response) {
  const limit = Math.min(parseInt(req.query.limit as string) || 10, 50);
  const days  = parseInt(req.query.days as string) || 30;
  const r = await pool.query(`
    SELECT
      p.id, p.name_en, p.name_ta, p.current_stock, p.min_stock_alert,
      SUM(bi.quantity)                 AS total_qty_sold,
      SUM(bi.quantity * bi.unit_price) AS total_revenue,
      (p.current_stock <= p.min_stock_alert) AS needs_restock
    FROM bill_items bi
    JOIN products p ON p.id = bi.product_id
    JOIN bills b    ON b.id = bi.bill_id
    WHERE b.created_at >= NOW() - ($2 || ' days')::INTERVAL
      AND b.payment_status = 'paid'
    GROUP BY p.id, p.name_en, p.name_ta, p.current_stock, p.min_stock_alert
    ORDER BY total_qty_sold DESC
    LIMIT $1
  `, [limit, days]);
  res.json(r.rows);
}
```
